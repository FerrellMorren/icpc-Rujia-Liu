P.S.这个代码算“位数”是不对的，因为t/=b的时候把一些东西扔掉了！
int g(int n,int b)
{
	int count=0;
	int i,t=1;
	for(i=1;i<=n;i++)
	{
		t*=i;
		while(t>=b)
		{
			t/=b;
			count++;
		}
	}
	count++;
	return count;
}
